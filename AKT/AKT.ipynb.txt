        "import copy\n",
        "from heapq import heappush, heappop\n",
        "\n",
        "# Giả sử puzzle n=3 (8-puzzle)\n",
        "n = 3\n",
        "\n",
        "# 4 vị trí dịch chuyển tương ứng: bottom, left, top, right\n",
        "rows = [1, 0, -1, 0]\n",
        "cols = [0, -1, 0, 1]\n",
        "\n",
        "# Tạo một lớp hàng đợi ưu tiên\n",
        "class priorityQueue:\n",
        "    def __init__(self):\n",
        "        self.heap = []\n",
        "\n",
        "    # Chèn một key mới\n",
        "    def push(self, key):\n",
        "        heappush(self.heap, key)\n",
        "\n",
        "    # Lấy phần tử có độ ưu tiên nhỏ nhất (min cost)\n",
        "    def pop(self):\n",
        "        return heappop(self.heap)\n",
        "\n",
        "    # Kiểm tra hàng đợi có trống không\n",
        "    def empty(self):\n",
        "        return not self.heap\n",
        "\n",
        "# Cấu trúc của một nút trong cây tìm kiếm\n",
        "class nodes:\n",
        "    def __init__(self, parent, mats, empty_tile_posi, costs, levels):\n",
        "        self.parent = parent\n",
        "        self.mats = mats\n",
        "        self.empty_tile_posi = empty_tile_posi\n",
        "        self.costs = costs\n",
        "        self.levels = levels\n",
        "\n",
        "    # Định nghĩa toán tử so sánh để hàng đợi ưu tiên hoạt động\n",
        "    def __lt__(self, nxt):\n",
        "        return self.costs < nxt.costs\n",
        "\n",
        "# Hàm tính toán số lượng các ô bị sai vị trí (misplaced tiles)\n",
        "def calculateCosts(mats, final) -> int:\n",
        "    count = 0\n",
        "    for i in range(n):\n",
        "        for j in range(n):\n",
        "            if (mats[i][j]) and (mats[i][j] != final[i][j]):\n",
        "                count += 1\n",
        "    return count\n",
        "\n",
        "# Hàm tạo một nút mới\n",
        "def newNodes(mats, empty_tile_posi, new_empty_tile_posi, levels, parent, final) -> nodes:\n",
        "    # Sao chép dữ liệu từ ma trận cha\n",
        "    new_mats = copy.deepcopy(mats)\n",
        "\n",
        "    # Di chuyển ô trống (swap vị trí)\n",
        "    x1 = empty_tile_posi[0]\n",
        "    y1 = empty_tile_posi[1]\n",
        "    x2 = new_empty_tile_posi[0]\n",
        "    y2 = new_empty_tile_posi[1]\n",
        "\n",
        "    new_mats[x1][y1], new_mats[x2][y2] = new_mats[x2][y2], new_mats[x1][y1]\n",
        "\n",
        "    # Tính toán chi phí (số ô sai vị trí)\n",
        "    costs = calculateCosts(new_mats, final)\n",
        "\n",
        "    new_nodes = nodes(parent, new_mats, new_empty_tile_posi, costs, levels)\n",
        "    return new_nodes\n",
        "\n",
        "# Hàm in ma trận N x N\n",
        "def printMatsrix(mats):\n",
        "    for i in range(n):\n",
        "        for j in range(n):\n",
        "            print(\"%d \" % (mats[i][j]), end=\" \")\n",
        "        print()\n",
        "\n",
        "# Kiểm tra tọa độ (x, y) có nằm trong ma trận không\n",
        "def isSafe(x, y):\n",
        "    return x >= 0 and x < n and y >= 0 and y < n\n",
        "\n",
        "# In đường đi từ nút gốc đến nút đích\n",
        "def printPath(root):\n",
        "    if root is None:\n",
        "        return\n",
        "    printPath(root.parent)\n",
        "    printMatsrix(root.mats)\n",
        "    print()\n",
        "\n",
        "# Hàm giải bài toán\n",
        "def solve(initial, empty_tile_posi, final):\n",
        "    pq = priorityQueue()\n",
        "\n",
        "    # Tạo nút gốc\n",
        "    costs = calculateCosts(initial, final)\n",
        "    root = nodes(None, initial, empty_tile_posi, costs, 0)\n",
        "\n",
        "    # Thêm nút gốc vào hàng đợi\n",
        "    pq.push(root)\n",
        "\n",
        "    while not pq.empty():\n",
        "        minimum = pq.pop()\n",
        "\n",
        "        # Nếu chi phí bằng 0, nghĩa là đã tìm thấy trạng thái đích\n",
        "        if minimum.costs == 0:\n",
        "            printPath(minimum)\n",
        "            return\n",
        "\n",
        "        # Tạo các nút con khả thi (4 hướng di chuyển)\n",
        "        for i in range(4): # Sử dụng 4 hướng thay vì range(n)\n",
        "            new_tile_posi = [\n",
        "                minimum.empty_tile_posi[0] + rows[i],\n",
        "                minimum.empty_tile_posi[1] + cols[i]\n",
        "            ]\n",
        "\n",
        "            if isSafe(new_tile_posi[0], new_tile_posi[1]):\n",
        "                child = newNodes(\n",
        "                    minimum.mats,\n",
        "                    minimum.empty_tile_posi,\n",
        "                    new_tile_posi,\n",
        "                    minimum.levels + 1,\n",
        "                    minimum,\n",
        "                    final\n",
        "                )\n",
        "                pq.push(child)\n",
        "\n",
        "# --- Thiết lập dữ liệu đầu vào và chạy chương trình ---\n",
        "\n",
        "initial = [\n",
        "    [1, 2, 3],\n",
        "    [5, 6, 0],\n",
        "    [7, 8, 4]\n",
        "]\n",
        "\n",
        "final = [\n",
        "    [1, 2, 3],\n",
        "    [5, 8, 6],\n",
        "    [0, 7, 4]\n",
        "]\n",
        "\n",
        "# Vị trí ô số 0 trong bảng initial là hàng 1, cột 2\n",
        "empty_tile_posi = [1, 2]\n",
        "\n",
        "solve(initial, empty_tile_posi, final)"
